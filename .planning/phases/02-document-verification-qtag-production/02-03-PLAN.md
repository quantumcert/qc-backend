         device: { id: string };
         input: VerifyTapInput;
         reason: DeniedReason;
         ctr?: number;
         cmacValid: boolean;
       }): Promise<void> { ... }

       const verdictByReason: Record<DeniedReason, TapVerdict | null> = {

<threat_model>
## Trust Boundaries

- Boundary A: NFC browser/client anonimo -> `/api/v1/scan`.
- Boundary B: dados criptograficos SDM -> `SDMVerifierService`.
- Boundary C: tentativas suspeitas -> `DeviceTapLog`.
- Boundary D: device desconhecido -> ausencia de log por falta de identificacao confiavel.

## Threats

| ID | Severity | Threat | Mitigation |
|----|----------|--------|------------|
| T-02-06 | high | tag clonada ou replay retorna erro publico mas nao deixa trilha forense | `logDeniedTap` para MAC, replay, relay e inactive quando device identificado |
| T-02-07 | high | logging de device desconhecido cria registros falsos ou impossiveis de atribuir | helper nao loga `DEVICE_NOT_FOUND` sem device confiavel |
| T-02-08 | medium | consumidores NFC quebram por mudanca de envelope `DENIED` | manter `status`, `reason`, `message` e testes de contrato |
| T-02-09 | medium | malformed scan vira 403/500 em vez de 400 | route test para `INVALID_INPUT` -> HTTP 400 |
</threat_model>
